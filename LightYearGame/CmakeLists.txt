function(CopyLibDirToTarget LIB_NAME TARGET_NAME)
add_custom_command(
    TARGET ${TARGET_NAME}
    POST_BUILD
    COMMAND 
    ${CMAKE_COMMAND} -E copy_directory
    $<TARGET_FILE_DIR:${LIB_NAME}>
    $<TARGET_FILE_DIR:${TARGET_NAME}>
)
endfunction()

CopyLibDirToTarget(${LIGHT_YEAR_ENGINE_TARGET_NAME} ${LIGHT_YEAR_GAME_TARGET_NAME})
